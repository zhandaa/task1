            if (input === '') {
              return '';
            }
          
            if (input[0] === input[0].toUpperCase()) {
              return input;
            } else {
              return input.split('').reverse().join('');
            }
          };

        const joinNumbersFromRange = (start, finish) => {
            let result = '';
            let i = start;
            while (i <= finish) {
              result += i;
              i += 1;
            }
            return result;
          };
          
          export default joinNumbersFromRange;

        const countChars = (sentence, char) => {
            const lowerCaseSentence = sentence.toLowerCase();
            const lowerCaseChar = char.toLowerCase();
            let count = 0;
          
            for (let i = 0; i < lowerCaseSentence.length; i++) {
              if (lowerCaseSentence[i] === lowerCaseChar) {
                count += 1;
              }
            }
          
            return count;
          };
          
          export default countChars;

        const encrypt = (message) => {
            let result = '';
          
            for (let i = 0; i < message.length - 1; i += 2) {
              result += message[i + 1] + message[i];
            }
          
            if (message.length % 2 !== 0) {
              result += message[message.length - 1];
            }
          
            return result;
          };
          
          export default encrypt;
